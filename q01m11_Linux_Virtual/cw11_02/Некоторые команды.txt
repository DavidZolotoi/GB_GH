lsblk - диски

ls -ailh

. [filename] - перезапустить файл в оперативке

pwd
cd - (вернуться на шаг назад по местуположению)

touch myFile - создать файл
cat myFile - посмотреть файл

mkdir -p myDir/f1 - создать по пути
rm -r myDir - рекурсивное удаление всего

mkdir myDir ; touch myDir/myFile - способ в одну строку

mv myFile myDir/myFile2 myDir/Dir - перенос всего в последнюю папку
myDir/* - означает взять всё что внутри и файлы и папки
при копировании нужно указывать рекурсию
cp -r myDir/* myDir/- аналогично

ln [filename] - жесткая - пока все они не удалены, из памяти не исчезнет - ссылается на область памяти
ln -s [filename] - символическая ссылка - ссылается на файл по пути файла
Изменение файла по любой ссылке, меняет все из них

> перезапись
>> дозапись

less - открыть содержимое на экране - можно крутить стрелками, выход q, после использования в терминале не будет следов файла
